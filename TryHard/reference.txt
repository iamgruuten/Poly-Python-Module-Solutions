#Removing duplicates in a list

names = ['lee', 'lee', 'tan', 'lee', 'lee', 'tan', 'kacey', 'oppo', 'apple', 'oppo']
store_name = names.copy()

for i in names:
    if names.count(i) > 1:

        names = store_name

#Sorting algorithm

def bubbleSort(alist):
    for passnum in range(len(alist)-1,0,-1):
        for i in range(passnum):
            if alist[i]>alist[i+1]:
                temp = alist[i]
                alist[i] = alist[i+1]
                alist[i+1] = temp

alist = [54,26,93,17,77,31,44,55,20]
bubbleSort(alist)
print(alist)
